#Defination of the class K_Means  
#Using this class we are implementing all the steps required  
#to be performed to generate a k-means model  
class K_Means:  
    def __init__(self, k=3, tol=0.001, max_iter=300):  
        self.k = k  
        self.tol = tol  
        self.max_iter = max_iter  
  
    #Function to fit the dummy mode on to the model  
    def fit(self,data):  
  
        self.centroids = {}  
  
        for i in range(self.k):  
            self.centroids[i] = data[i]  
  
        for i in range(self.max_iter):  
            self.classifications = {}  
  
            for i in range(self.k):  
                self.classifications[i] = []  
  
            for featureset in data:  
                distances = [np.linalg.norm(featureset-self.centroids[centroid]) for centroid in self.centroids]  
                classification = distances.index(min(distances))  
                self.classifications[classification].append(featureset)  
  
            prev_centroids = dict(self.centroids)  
  
            for classification in self.classifications:  
                self.centroids[classification] = np.average(self.classifications[classification],axis=0)  
  
            optimized = True  
  
            for c in self.centroids:  
                original_centroid = prev_centroids[c]  
                current_centroid = self.centroids[c]  
                if np.sum((current_centroid-original_centroid)/original_centroid*100.0) > self.tol:  
                    np.sum((current_centroid-original_centroid)/original_centroid*100.0)  
                    optimized = False  
  
            if optimized:  
                break  
  
    #Function to generate prediction result  
    def predict(self,data):  
        distances = [np.linalg.norm(data-self.centroids[centroid]) for centroid in self.centroids]  
        classification = distances.index(min(distances))  
        return classification

# Function displaying the menu, returns the choice
def menu_choice():
    return "add"
    return input("Please select from below:\n"
                 "Option 1 - add\n"
                 "Option 2 - subtract\n"
                 "Option 3 - multiply\n"
                 "Option 4 - divide\n"
                 "Option 5 - quit\n-> ").lower()  # Return the answer in lower case

actions = {
    ("1", "add"): add_num,
    ("2", "subtract"): subtract_num,
    ("3", "multiply"): multiply_num,
    ("4", "divide"): divide_num,
}


# Main function that runs the program.
# Continuously gets the users choice using menu_choice(), checks that the user does not want to quit. (5/quit)
# If not, the users is asked for 2 numbers
# Then, all the keys
# Loop ends when the user enters 5 or quit
def run():
    # Display welcome on first entry
    print("Welcome to my calculator!")
    # Get the users choice
    choice = menu_choice()
    # Loop until the user wants to quit
    count=0
    while choice not in ("5", "quit"):
        
        # Get 2 numbers from the user
        num_1 = 3  #float(input("Enter 1st number -> "))
        num_2 = 4   #float(input("Enter 2nd number -> "))

        # Determine what the user wants to do
        # Loop through each key, find the tuple containing the users choice.
        # Then access and print the appropriate function
        for key in actions.keys():
            if choice in key:
                print(actions[key](num_1, num_2))
                break
        print("______________________________________________")

        count+=1
        if count>10:
            choice = "5"


# Run the program
run()

# Main function that runs the program.
# Continuously gets the users choice using menu_choice(), checks that the user does not want to quit. (5/quit)
# If not, the users is asked for 2 numbers
# Then depending on what the user wants to do the appropriate function is called
# Any unknown function does nothing except notify the user that it is incorrect
# Loop ends when the user enters 5 or quit
def run():
    # Display welcome on first entry
    print("Welcome to my calculator!")
    # Get the users choice
    choice = menu_choice()
    # Loop until the user wants to quit
    count = 0
    while choice not in ("5", "quit"):
        
        # Get 2 numbers from the user
        num_1 = 3 # float(input("Enter 1st number -> "))
        num_2 = 4 #float(input("Enter 2nd number -> "))

        # Determine what the user wants to do
        if choice in ("1", "add"):
            print(add_num(num_1, num_2))
        elif choice in ("2", "subtract"):
            print(subtract_num(num_1, num_2))
        elif choice in ("3", "multiply"):
            print(multiply_num(num_1, num_2))
        elif choice in ("4", "divide"):
            print(divide_num(num_1, num_2))
        else:
            print("Incorrect input. Try again.")
        print("______________________________________________")

        # Get a new choice from the user
        count+=1
        if count>10:
            break
        
        


# Run the program
